- JDBC 프로그램 구현
```java
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

public class Practice {

    static String JDBC_URL = "jdbc:oracle:thin:@localhost:1521:xe";
    static String JDBC_USER = "system";
    static String JDBC_PASSWORD = "1234";

    public static void main(String[] args) {

        Connection con = null;
        PreparedStatement pstmt = null;
        ResultSet rs = null;

        String sql;
        String temp;
        String keyword = "여행";

        try {
            //  1. (JDBC)Driver Loading
            Class.forName("oracle.jdbc.driver.OracleDriver");
            System.out.println("Driver loaded");

            //  2. database 연결
            con = DriverManager.getConnection(JDBC_URL, JDBC_USER, JDBC_PASSWORD);
            System.out.println("database connected");

            //  3. 쿼리작성
            sql = "select * from book where btitle like ?";

            pstmt = con.prepareStatement(sql);

            pstmt.setString(1, "%" + keyword + "%");

            //  4. 쿼리 실행
            rs = pstmt.executeQuery();

            //  5. 결과 처리
            while (rs.next()) {
                temp = rs.getString("bisbn") + " ";
                temp += rs.getString("btitle") + " ";
                temp += rs.getString("bauthor") + " ";
                temp += rs.getString("bprice") + " ";
                System.out.println(temp);
            }

        } catch(Exception e){
            System.out.println("Error: " + e);
        } finally {
            //  6. 리소스 해제
            try{
                rs.close();
                pstmt.close();
                if( con != null ){
                    con.close();
                }
                System.out.println("database closed");

            } catch(Exception e){
                System.out.println("Error: " + e);
            }

        }
    }
}
```

- Transaction
```java
package first;

import java.sql.*;

public class TransactionTest {
    static String JDBC_URL = "jdbc:oracle:thin:@localhost:1521:xe";
    static String JDBC_USER = "system";
    static String JDBC_PASSWORD = "1234";

    public static void main(String[] args) {
        Connection con = null;
        PreparedStatement pstmt = null;
        ResultSet rs = null;

        String sql = null;

        try {
            //  1. Driver loading
            Class.forName("oracle.jdbc.driver.OracleDriver");
            System.out.println("Driver: driver loaded success");
            
            //  2. Connection 연결 ( 기본적으로 auto commit mode 로 connection 생성 )
            con = DriverManager.getConnection(JDBC_URL, JDBC_USER, JDBC_PASSWORD);
            
            //  transaction 설정
            con.setAutoCommit(false);   //  transaction 시작
            // 명시적으로 commit() rollback() 호출 되면 transaction 종료
            // connection 정상종료 되면 commit()으로 간주
            System.out.println("Database: Database connection success");
            //  3.SQL 작성
            sql = "insert into MEMBER_TBL values(?,?)";
            pstmt = con.prepareStatement(sql);
            pstmt.setString(1, "2");
            pstmt.setString(2, "shin");
            
            //  4. SQL 구문 실행
            int result = pstmt.executeUpdate(); //영향받은 row 수

            //  5. 결과처리
            if(result == 1) {
                System.out.println("Insert success");
            } else {
                System.out.println("Insert failed");
            }
            
            //  Transaction 종료
            con.commit();


        } catch (Exception e) {
            System.out.println("error: " + e.getMessage());
        } finally {
            //  6. 리소스 해제
            try {
                pstmt.close();
                if (con != null)
                    con.close();
                System.out.println("Database: Database closed success");
            } catch (Exception e2) {
                System.out.println("error: " + e2.getMessage());
            }
        }
    }
}
```
