package assignment;

import java.util.Scanner;

public class Test01 {
    //2개의 정수형 데이터(1부터10)를 입력 받아 첫 번째 입력한 값을 분자로 하고 두 번째 입력

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double num1 = sc.nextInt();
        double num2 = sc.nextInt();

        double ans = num1%num2;
        if(ans>1){
            System.out.println("나머지가 1보다 크다!");
        } else {
            System.out.println("나머지가 1보다 작거나 같다!");
        }

    }
}

package assignment;

import java.util.Scanner;

public class Test02 {
    //정수형 데이터 1자리(범위 5~10)를 입력 받아 1부터 입력 받은 숫자까지의 곱을 구한다.
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();

        int ans = 1;

        for (int i = 1; i <= n; i++) {
            ans = i * ans;
        }
        System.out.println(ans);

    }
}

package assignment;

public class Test04 {
    //세 개의 주사위를 던졌을 때, 눈의 곱이 3의 배수인 값을 출력하는 프로그램을 작성하시오.

    public static void main(String[] args) {
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 6; j++) {
                for (int k = 0; k < 6; k++) {
                    if( ((i + 1)*(j + 1)*(k + 1))%3 == 0 )
                        System.out.println((i+1)+"*"+(j+1)+"*"+(k+1)+"="+((i+1)*(j+1)*(k+1)));
                }
            }
        }
    }

package assignment;

public class Test05 {
    //3개의 1자리(1~9) 정수형 데이터를 받아 최대 값과 최소 값을 화면에 출력 한다
    public static void main(String[] args) {


        int[] argsInt = new int[args.length];
        for (int i = 0; i < args.length; i++) {
            argsInt[i] = Integer.parseInt(args[i]);
        }

        int min = argsInt[0];
        int max = argsInt[0];

        for (int i = 1; i < args.length; i++) {
            min = Math.min(min, argsInt[i]);
            max = Math.max(max, argsInt[i]);
        }

        System.out.println("최대값: " + max);
        System.out.println("최소값: " + min);
    }
}
